How to set up 6to5 with Gulp:

$ npm install --save-dev gulp-6to5
var gulp = require('gulp');
var to5 = require('gulp-6to5');

return gulp.task('build', function() {
    gulp.src('src/app.js')
        .pipe(to5())
        .pipe(gulp.dest('dist'));
});
$ gulp build

How to set up Traceur with Gulp:

$ npm install --save-dev gulp-traceur
var gulp = require('gulp');
var traceur = require('gulp-traceur');

gulp.task('default', function () {
    return gulp.src('src/app.js')
        .pipe(traceur({sourceMap: true}))
        .pipe(gulp.dest('dist'));
});
$ gulp build

How to set up 6to5 with Grunt:

$ npm install --save-dev grunt-6to5
grunt.loadNpmTasks('grunt-6to5');

grunt.initConfig({
    '6to5': {
        options: {
            sourceMap: true
        },
        dist: {
            files: {
                'dist/app.js': 'src/app.js'
            }
        }
    }
});

grunt.registerTask('build', ['6to5']);
$ grunt build

How to set up Traceur with Grunt:

$ npm install --save-dev grunt-traceur
grunt.loadNpmTasks('grunt-traceur');

grunt.initConfig({

How to set up esnext with Grunt:

$ npm install --save-dev grunt-esnext
grunt.loadNpmTasks('grunt-esnext');

grunt.initConfig({
  esnext: {
    options: {
      includeRuntime: true
    },
    dist: {
      files: {
        'dist/app.js': 'src/app.js'
      }
    }
  }
});

grunt.registerTask('build', ['esnext']);
$ grunt build

How to set up 6to5 with Browserify:

$ npm install --save-dev 6to5-browserify
// build.js
var fs = require('fs');
var browserify = require('browserify');
var to5 = require('6to5-browserify');

browserify({ debug: true })
    .transform(to5)
    .require('./src/app.js', { entry: true })
    .bundle()
    .pipe(fs.createWriteStream('./dist/app.js'));
$ node build.js

How to set up Traceur with Browserify:

$ npm install --save-dev es6ify
// build.js
var fs = require('fs');
var browserify = require('browserify');
var es6ify = require('es6ify');

browserify({ debug: true })
    .add(es6ify.runtime)
    .transform(es6ify)
    .require('./src/app.js', { entry: true })
    .bundle()
    .pipe(fs.createWriteStream('./dist/app.js'));
$ node build.js

How to set up 6to5 with Command Line:

$ npm install --global 6to5
$ 6to5 src/app.js -s -o dist/app.js

How to set up Traceur with Command Line:

$ npm install --global traceur
$ traceur src/app.js --source-maps file --out dist/app.js

How to set up esnext with Command Line:

$ npm install --global esnext
$ esnext src/app.js --source-maps -o dist/app.js
